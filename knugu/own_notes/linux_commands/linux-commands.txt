~ - коренева папка.

mkdir project - створити папку project.

cd project - зайти в папку project.

cd .. - вийти з папки.

cd - вийти в кореневу папку.

cd d:\Project/dir1 - зайти в папку dir1 (шлях).

touch readme.txt - створити файл readme.txt

ls - виводить список файлів і папок в даній директорії.

ls -a - виводить список файлів і папок в даній директорії з прихованими.

ls -l - виводить список файлів і папок в даній директорії в більш широкому форматі.

man + command - інсирукція по команді (manual).

rm - видалити файл.

rmdir - видалити порожню папку.

rm -r nameDir - видалити папку з вмістом.

grep "ключове слово(фраза)" "ім'я файлу" - виводить всі стрічки, де є ключове слово(фраза).

pwd - (print working directory) - показує дану директорію.

clear (ctrl + l) - очищує екран.
